void CMFCApplication9Dlg::OnMouseMove(UINT nFlags, CPoint point)
{
	CClientDC dc(this);
	dc.Rectangle(0, 0, point.x, point.y);
	CDialogEx::OnMouseMove(nFlags, point);


}

void CMFCApplication9Dlg::OnMouseMove(UINT nFlags, CPoint point)
//UINT == unsignedinteger
{
	if (nFlags & MK_LBUTTON) {
	CClientDC dc(this);
	CPen pen(PS_SOLID, 10, RGB(0, 0, 255));
	dc.SelectObject(&pen);
	dc.MoveTo(point);
	dc.LineTo(point);}
	CDialogEx::OnMouseMove(nFlags, point);
}
