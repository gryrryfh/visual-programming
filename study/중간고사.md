COLORREF col;
CPoint oPnt;
void CMFCApplication4Dlg::OnMouseMove(UINT nFlags, CPoint point)
{
	if (nFlags & MK_LBUTTON) {
		CPen pen(PS_SOLID, 16, col);
		CClientDC dc(this);
		dc.SelectObject(&pen);
		dc.MoveTo(oPnt);
		dc.LineTo(point);

	}
	oPnt = point;
	CDialogEx::OnMouseMove(nFlags, point);
}

void CMFCApplication4Dlg::OnBnClickedButton1()
{
	CColorDialog dlg;
	if (dlg.DoModal() == IDOK) {
		col = dlg.GetColor();
	}
}
