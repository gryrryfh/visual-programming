dialog가 아닌 다른 것을 사용하려면 AfxMessageBox를 사용해야됨
```MFC
BOOL CMy2Dlg::PreTranslateMessage(MSG* pMsg)
{
	if (pMsg->message == WM_KEYDOWN) {
		if (pMsg->wParam == 'A')
			AfxMessageBox(L"A키 눌림");

		if (pMsg->wParam == 'B') {
			CClientDC dc(this);
			dc.Rectangle(20, 20, 100, 100);
		}
	}

	return CDialogEx::PreTranslateMessage(pMsg);
}
```
